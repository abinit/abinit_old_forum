program mpitag
use mpi
implicit none
integer :: err

logical flag


call MPI_INIT(err)
call MPI_COMM_GET_ATTR(MPI_COMM_WORLD, MPI_TAG_UB, my_tag_ub, flag, err)
if (flag .eqv. .true.) then
print *, "Got tag ub:", my_tag_ub

print *, "Couldn't find tag ub!"
endif
call MPI_FINALIZE(err)
