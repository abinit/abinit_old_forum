udtset 1 1

# Definition of the unit cell

  1 0 0

  0 0 1

# Definition of the atoms
natom 5

xred
  0.0 0.0 0.0                   # Sr sits at cubic site
  0.5 0.5 0.5                   # Ti sits at body-center site
  0.0 0.5 0.5                   # O sits at face-centered sites
  0.5 0.0 0.5
  0.5 0.5 0.0

ecut1? 14

kptopt 1
